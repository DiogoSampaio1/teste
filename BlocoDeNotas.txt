<script>
  let scannerAtivo = false;

  function iniciarScanner() {
    const readerDiv = document.getElementById("reader");

    if (scannerAtivo) return; // Evita mÃºltiplas ativaÃ§Ãµes

      }

      resultadoDiv.innerText = `Erro ao acessar cÃ¢mera: ${err}`;
